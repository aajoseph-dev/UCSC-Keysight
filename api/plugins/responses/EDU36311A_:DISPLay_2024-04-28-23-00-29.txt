Based on the provided context and the SCPI command documentation for the `EDU36311A` power supply, I will create a Python class that encapsulates a subset of the SCPI commands related to the display functions of the device. The class will enable or disable the display and query its status via the `:DISPlay:ENABle` SCPI command.

```python
import pyvisa

class EDU36311APowerSupply:
    def __init__(self, resource_name):
        # Initialize VISA resource manager
        self.rm = pyvisa.ResourceManager()
        self.instrument = self.rm.open_resource(resource_name)

    def enable_display(self, enable=True):
        """
        Enables or disables the display of the power supply.
        
        :param enable: Boolean indicating whether to enable (True) or disable (False) the display.
        """
        command = ":DISPlay:ENABle " + ("ON" if enable else "OFF")
        self.instrument.write(command)

    def query_display_enable(self):
        """
        Queries the display status of the power supply.
        
        :return: Boolean indicating the status of the display (True for enabled, False for disabled).
        """
        response = self.instrument.query(":DISPlay:ENABle?")
        return True if response.strip().upper() == "ON" else False

    def close(self):
        """
        Closes the connection to the power supply.
        """
        self.instrument.close()

# Example usage:
if __name__ == "__main__":
    # Substitute 'USB0::0x2A8D::0x1202::MY1234567::INSTR' with the actual VISA address of your device
    power_supply = EDU36311APowerSupply('USB0::0x2A8D::0x1202::MY1234567::INSTR')
    
    try:
        # Enable the power supply display
        power_supply.enable_display(True)
        print("Display enabled.")

        # Query and print the display status
        display_status = power_supply.query_display_enable()
        print(f"Display status: {'Enabled' if display_status else 'Disabled'}")
        
        # Disable the power supply display
        power_supply.enable_display(False)
        print("Display disabled.")
    finally:
        # Ensure the connection is closed regardless of errors
        power_supply.close()
```
